    (add-1 zero)   
    -> (lambda (f) (lambda (x) (f ((zero f) x))))    
    -> (lambda (f) (lambda (x) (f (((lambda (f) (lambda (x) x)) f) x))))    
    -> (lambda (f) (lambda (x) (f ((lambda (x) x) x))))   
    -> (lambda (f) (lambda (x) (f x)))

      (lambda (f) (lambda (x) (f x)))  
    )

    (define two
      (lambda (f) (lambda (x) (f (f x))))
