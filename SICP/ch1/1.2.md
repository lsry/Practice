    (define (pascal n x)
      (cond ((or (< x 1) (> x n)) 0)
            ((= n 1) 1)
        (else
          (+ (pascal (- n 1) x) (pascal (- n 1) (- x 1)))
        )
      )
    )

#### 1.14  
图略   
增长步数：O(n^k)    
不妨先假设只有一种1分的，则：     
f(1) = 3  
f(n) = f(n-1) + 1 = f(n-2) + 2 = ... = f(1) + 2(n-1) = O(n)          
现在加入一种5分的，令 d = n/5  则：      
g(n) = f(n) + g(n-5) + 1  
= O(n) + f(n-5) + g(n-10) + 2   
= O(n) + O(n-5) + f(n-10) + g(n-15) + 3   
= ...     
= O(nd) + d  = O(n^2)   
....  
加入k种后： O(n^k)              
空间：O(n)
